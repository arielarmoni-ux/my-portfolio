// לוגיקת כפתור חזרה למעלה - וודא שזה קיים בתוך פונקציית ה-init או תחתיה const btt = document.getElementById("back-to-top"); window.onscroll = () => { if (window.scrollY > 400) { btt.classList.add("visible"); } else { btt.classList.remove("visible"); } }; btt.onclick = (e) => { e.preventDefault(); window.scrollTo({ top: 0, behavior: 'smooth' }); };
